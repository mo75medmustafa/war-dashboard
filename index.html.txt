    <script>
        async function fetchDashboard() {
            try {
                const res = await fetch('/api/dashboard');
                const data = await res.json();
                
                document.getElementById('mil-val').innerText = data.military_escalation + '%';
                document.getElementById('mil-trend').innerText = data.military_trend;
                document.getElementById('dur-val').innerText = data.war_duration + '%';
                document.getElementById('dur-trend').innerText = data.war_duration_trend;

                document.getElementById('news-list').innerHTML = data.official_news.map(n => 
                    `<li><strong>${n.source}:</strong> ${n.title}</li>`
                ).join('');
            } catch (e) { console.error("خطأ:", e); }
        }
        fetchDashboard();
    </script>
