import { createClient } from '@supabase/supabase-js';

// استدعاء المفاتيح من إعدادات الاستضافة (هنعملها في الخطوة الجاية)
const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);

export default async function handler(req, res) {
  try {
    const { range } = req.query;

    // جلب البيانات من الجداول اللي عملناها في الخطوة الأولى
    const { data: events } = await supabase.from('events').select('*');
    const { data: news } = await supabase.from('official_news').select('*').order('id', {ascending: false}).limit(5);
    const { data: prices } = await supabase.from('regional_prices').select('*').limit(10);

    // حساب نسبة التصعيد (معادلة بسيطة)
    let totalScore = 0;
    events?.forEach(ev => totalScore += (ev.severity * 10)); 
    const escalationIndex = events?.length ? Math.min(100, totalScore / (events.length)) : 0;

    // الرد النهائي اللي هيروح للوحة
    res.status(200).json({
      military_escalation: Math.round(escalationIndex),
      military_trend: escalationIndex > 60 ? "تصعيد حاد" : "هدوء حذر",
      official_news: news || [],
      regional_prices: prices || [],
      war_duration: Math.round(escalationIndex * 0.8),
      war_duration_trend: "توقعات باستمرار العمليات"
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
}
